:: change default shell of user
    sudo useradd -D -s /new/shell # 1st way
    chsh -s /new/shell USERNAME    # 2nd way

:: check cron jobs
    for user in $(cat /etc/passwd | cut -f1 -d: ); do echo $user; crontab -u $user -l; done

:: check for users who have login shell only
    for user in $(cat /etc/passwd | grep sh$ | cut -f1 -d: ); do echo $user; crontab -u $user -l; done

:: checking processes
    ps -f
    ps auxfww # username, TTY, wide output

:: finding SUID
    find / -type f -perm -u=s 2>/dev/null
